** Installation

#+BEGIN_SRC shell
$ git clone -b develop --single-branch https://github.com/hlissner/doom-emacs.git .emacs.d
$ git clone -b develop --single-branch https://github.com/jwintz/doom.git .doom.d
$ ln -s ~/.doom.d/init.doom.el ~/.emacs.d/init.el
$ cd ~/.emacs.d
$ make
$ make doctor
#+END_SRC
